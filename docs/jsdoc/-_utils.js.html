/**
 * Shorthand for document & body.
 */
const doc = document,
  body = doc.body;

/**
 * Shorthand for console log.
 */
const log = console.log.bind(doc);

/**
 * Fast change class.
 * @param {object} element - Manipulation element
 * @param {string} className - Class name
 */
const addClass = (element, className) => {
    element.classList.add(className);
  },
  removeClass = (element, className) => {
    element.classList.remove(className);
  };

/**
 * Fast select item(s).
 * @param {string} selector - Css selector
 * @param {object} scope - The parent of the item(s) to be selected
 * @return {object}
 */
const select = (selector, scope) => {
    scope = scope ? scope : document;
    return scope.querySelector(selector);
  },
  selects = (selector, scope) => {
    scope = scope ? scope : document;
    return scope.querySelectorAll(selector);
  };

/**
 * Setting scroll status function.
 * @param {integer} status
 */
const scroll = (status = 1) => {
  status === 0
    ? (body.style.overflowY = 'hidden')
    : (body.style.overflowY = 'auto');
};

/**
 * Get element position.
 * @param {object} obje
 */
const position = (obje) => {
  const box = obje.getBoundingClientRect(),
    docEl = doc.documentElement;

  let scrollTop = window.pageYOffset || docEl.scrollTop || body.scrollTop,
    scrollLeft = window.pageXOffset || docEl.scrollLeft || body.scrollLeft,
    clientTop = docEl.clientTop || body.clientTop || 0,
    clientLeft = docEl.clientLeft || body.clientLeft || 0,
    top = box.top + scrollTop - clientTop,
    left = box.left + scrollLeft - clientLeft;

  return { top: Math.round(top), left: Math.round(left) };
};
